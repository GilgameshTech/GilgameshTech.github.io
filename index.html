public interface Displayable {
    void displayDetails();
}
public abstract class LibraryItem implements Displayable {
    private static int totalItems = 0;
    protected String id;
    protected String title;

    public LibraryItem(String id, String title) {
        this.id = id;
        this.title = title;
        totalItems++;
    }
    
    public abstract String getItemType();
    
    public static int getTotalItems() {
        return totalItems;
    }
    
    @Override
    public void displayDetails() {
        System.out.println("Type  : " + getItemType());
        System.out.println("ID    : " + id);
        System.out.println("Title : " + title);
    }
}
import java.util.ArrayList;
import java.util.List;

public class Shelf<T extends LibraryItem> {
    private List<T> itemsOnShelf = new ArrayList<>();
    private String shelfName;

    public Shelf(String shelfName) {
        this.shelfName = shelfName;
    }

    public void addItems(T... items) {
        for (T item : items) {
            itemsOnShelf.add(item);
        }
    }

    public void displayShelfContents() {
        System.out.println("=== Isi Rak: " + shelfName + " ===");
        itemsOnShelf.forEach(item -> {
            item.displayDetails();
            System.out.println("---");
        });
    }
}
public class Book extends LibraryItem {
    private String author;

    public Book(String id, String title, String author) {
        super(id, title);
        this.author = author;
    }

    @Override
    public String getItemType() {
        return "Book";
    }

    @Override
    public void displayDetails() {
        super.displayDetails();
        System.out.println("Author: " + author);
    }
}
public class Magazine extends LibraryItem {
    private int issueNumber;

    public Magazine(String id, String title, int issueNumber) {
        super(id, title);
        this.issueNumber = issueNumber;
    }

    @Override
    public String getItemType() {
        return "Magazine";
    }

    @Override
    public void displayDetails() {
        super.displayDetails();
        System.out.println("Issue : " + issueNumber);
    }
}
public class LibraryManager {
    public static final String LIBRARY_NAME = "The Digital Athenaeum";

    public static void main(String[] args) {
        System.out.println("Selamat datang di " + LIBRARY_NAME);
        System.out.println("=================================");

        Shelf<Book> bookShelf = new Shelf<>("Fiksi & Petualangan");
        Shelf<Magazine> magazineRack = new Shelf<>("Sains & Pengetahuan");
        
        bookShelf.addItems(
            new Book("BK-001", "The Gobbit", "J.R.R. Tokek"),
            new Book("BK-002", "Garry Potter", "Adam Yes")
        );

        magazineRack.addItems(
            new Magazine("MG-001", "National Geographic", 142),
            new Magazine("MG-002", "Scientific Lemurian", 329)
        );

        bookShelf.displayShelfContents();
        magazineRack.displayShelfContents();
        
        System.out.println("=================================");
        System.out.println("Total koleksi di sistem: " + LibraryItem.getTotalItems());
    }
}
