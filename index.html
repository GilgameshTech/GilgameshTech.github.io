// Aksi.java - Mendefinisikan kontrak atau kemampuan dasar entitas
public interface Aksi {
    void tampilkanStatus();
    String getTipeEntitas();
}
// Entitas.java - Kelas dasar abstrak untuk semua entitas
public abstract class Entitas implements Aksi {
    protected String nama;
    protected int level;

    public Entitas(String nama, int level) {
        this.nama = nama;
        this.level = level;
    }

    // Metode abstract, harus diimplementasikan oleh anak kelas
    public abstract String getTipeEntitas();

    @Override
    public void tampilkanStatus() {
        System.out.println("Nama   : " + this.nama);
        System.out.println("Level  : " + this.level);
        System.out.println("Tipe   : " + this.getTipeEntitas());
    }
}
// Pahlawan.java - Kelas konkret turunan dari Entitas
public class Pahlawan extends Entitas {
    private String senjata;

    public Pahlawan(String nama, int level, String senjata) {
        super(nama, level);
        this.senjata = senjata;
    }

    @Override
    public String getTipeEntitas() {
        return "Pahlawan";
    }

    @Override
    public void tampilkanStatus() {
        super.tampilkanStatus();
        System.out.println("Senjata: " + this.senjata);
    }
}
// Bot.java - Kelas konkret turunan dari Entitas
public class Bot extends Entitas {
    private String idBot;

    public Bot(String nama, int level, String idBot) {
        super(nama, level);
        this.idBot = idBot;
    }

    @Override
    public String getTipeEntitas() {
        return "Bot";
    }

    @Override
    public void tampilkanStatus() {
        super.tampilkanStatus();
        System.out.println("Bot ID : " + this.idBot);
    }
}
// ManajemenEntitas.java - Titik masuk utama aplikasi
import java.util.ArrayList;
import java.util.List;

public class ManajemenEntitas {
    public static void main(String[] args) {
        // Polimorfisme: Pahlawan dan Bot disimpan sebagai Entitas
        Entitas pahlawan1 = new Pahlawan("Aragorn", 50, "Anduril");
        Entitas bot1 = new Bot("Sentry-01", 99, "BOT-A8B3");

        List<Entitas> daftarEntitas = new ArrayList<>();
        daftarEntitas.add(pahlawan1);
        daftarEntitas.add(bot1);

        for (Entitas entitas : daftarEntitas) {
            entitas.tampilkanStatus();
            System.out.println("-----------------");
        }
    }
}
