// Action.java - Defines the basic contract for entities.
public interface Action {
    void displayStatus();
    String getEntityType();
}
// Entity.java - The abstract base class for all entities.
public abstract class Entity implements Action {
    protected String name;
    protected int level;

    public Entity(String name, int level) {
        this.name = name;
        this.level = level;
    }

    // Abstract method, must be implemented by subclasses.
    public abstract String getEntityType();

    @Override
    public void displayStatus() {
        System.out.println("Name   : " + this.name);
        System.out.println("Level  : " + this.level);
        System.out.println("Type   : " + this.getEntityType());
    }
}
// Hero.java - A concrete class derived from Entity.
public class Hero extends Entity {
    private String weapon;

    public Hero(String name, int level, String weapon) {
        super(name, level);
        this.weapon = weapon;
    }

    @Override
    public String getEntityType() {
        return "Hero";
    }

    @Override
    public void displayStatus() {
        super.displayStatus();
        System.out.println("Weapon: " + this.weapon);
    }
}
// Bot.java - A concrete class derived from Entity.
public class Bot extends Entity {
    private String botId;

    public Bot(String name, int level, String botId) {
        super(name, level);
        this.botId = botId;
    }

    @Override
    public String getEntityType() {
        return "Bot";
    }

    @Override
    public void displayStatus() {
        super.displayStatus();
        System.out.println("Bot ID : " + this.botId);
    }
}
// EntityManager.java - The main entry point of the application.
import java.util.ArrayList;
import java.util.List;

public class EntityManager {
    public static void main(String[] args) {
        // Polymorphism: Hero and Bot are stored as Entity.
        Entity hero1 = new Hero("Aragorn", 50, "Anduril");
        Entity bot1 = new Bot("Sentry-01", 99, "BOT-A8B3");

        List<Entity> entityList = new ArrayList<>();
        entityList.add(hero1);
        entityList.add(bot1);

        for (Entity entity : entityList) {
            entity.displayStatus();
            System.out.println("-----------------");
        }
    }
}
